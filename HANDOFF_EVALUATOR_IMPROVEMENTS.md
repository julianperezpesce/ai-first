### Priority 4: Intelligent Caching

**Goal**: Only re-evaluate files that changed.

**Implementation**:
```typescript
// src/core/evaluation/evaluation-cache.ts

interface CacheEntry {
  fileHash: string;  // MD5
  timestamp: number;
  evaluation: EvaluationResult;
  ttl: number;
}

class EvaluationCache {
  private cache: Map<string, CacheEntry>;
  
  shouldReevaluate(filePath: string): boolean {
    const currentHash = calculateMD5(filePath);
    const cached = this.cache.get(filePath);
    
    if (!cached) return true;
    if (currentHash !== cached.fileHash) return true;
    if (Date.now() - cached.timestamp > cached.ttl) return true;
    
    return false;
  }
  
  async getOrEvaluate(
    filePath: string, 
    evaluator: () => Promise<EvaluationResult>
  ): Promise<EvaluationResult> {
    if (!this.shouldReevaluate(filePath)) {
      return this.cache.get(filePath)!.evaluation;
    }
    
    const result = await evaluator();
    this.cache.set(filePath, {
      fileHash: calculateMD5(filePath),
      timestamp: Date.now(),
      evaluation: result,
      ttl: 24 * 60 * 60 * 1000 // 24 hours
    });
    
    return result;
  }
}
```

**Goal**: Replace simple Set() deduplication with semantic grouping.

async function synthesizeWithLLM(improvements: string[]): Promise<string[]> {
  const prompt = `
    Agrupa y sintetiza estas mejoras sugeridas por 3 modelos:
    ${improvements.map((imp, i) => `${i + 1}. ${imp}`).join('\n')}
    
    Tareas:
    1. Agrupa similares semánticamente
    2. Elimina duplicados
    3. Prioriza por impacto
    4. Máximo 5 recomendaciones
    
    Responde con JSON: { "consolidated": ["..."] }
  `;
  
  const response = await callLLM(prompt);
  return response.consolidated;
}

### Priority 7: CI/CD Integration

**Goal**: Run evaluation automatically on PRs.

**GitHub Actions Workflow**: `.github/workflows/evaluate-context.yml`
```yaml
name: Evaluate AI Context

on:
  pull_request:
    paths:
      - 'src/**'
      - 'test-projects/**'

jobs:
  evaluate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
          
      - name: Install dependencies
        run: npm ci
        
      - name: Build
        run: npm run build
        
      - name: Generate ai-context for test projects
        run: |
          node dist/index.js init --root test-projects/salesforce-cli
          node dist/index.js init --root test-projects/go-microservice
          # ... other projects
          
      - name: Run evaluation
        run: npm run evaluate:ci
        env:
          EVALUATION_THRESHOLD: 3.5
          OPENCODE_API_KEY: ${{ secrets.OPENCODE_API_KEY }}
          
      - name: Upload evaluation report
        uses: actions/upload-artifact@v3
        with:
          name: evaluation-report
          path: evaluation-report.md
```
